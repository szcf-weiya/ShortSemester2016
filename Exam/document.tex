	\begin{lstlisting}
	syms x
	>> syms x
	>> eqn = x^3+7*x+1 == 0;
	>> solx = solve(eqn,x)
	
	solx =
	
	root(z^3 + 7*z + 1, z, 1)
	root(z^3 + 7*z + 1, z, 2)
	root(z^3 + 7*z + 1, z, 3)
	
	>> x = (-10:0.001:10);
	>> y = x.^3+7*x+1;
	>> plot(x,y)
	>> S_vpa = vpa(solx)

	-0.14244425060428649657171963060493

	>> hold on;
	>> line([-10,10],[0,0]);
